- create table Course(courseId int primary key auto_increment, courseName varchar(50), fees int, duration varchar(20));

--Creating Sudent table:

- create table Student(studentId int primary key, studentname varchar(50), city varchar(50), age int, courseId int, foreign key(courseId) references Course(courseId));
